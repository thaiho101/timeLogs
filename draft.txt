  <button class="btn" onclick="deleteLast()">←</button>

<script>
  let code = "";

  function press(num) {
    if (code.length < 4) {
      code += num;
      updateDisplay();
    }
  }

  function updateDisplay() {
    document.getElementById('display').value = '*'.repeat(code.length);
  }

  function clearCode() {
    code = "";
    updateDisplay();
  }

  function deleteLast() {
    code = code.slice(0, -1);
    updateDisplay();
  }
</script>
